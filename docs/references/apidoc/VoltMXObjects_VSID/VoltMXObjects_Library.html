        jsonSort
        		
        							
        Public Function jsonSort(dataObj As JsonObject, lbl1 As String, lbl1Desc As Boolean, lbl2 As String, lbl2Desc As Boolean, lbl3 As String, lbl3Desc As Boolean, delim As String) As JsonObject
        			        				
        Given an array of JSON objects, returns a sorted array of JSON objects. Uses VoltScript Collections internally as a dependency.
        	
        	idx	Argument	Data Type	Attributes	Comments
        1dataObjJsonObjectJSON Object to be sorted 
        2lbl1StringFirst label on which to sort 
        3lbl1DescBooleanTrue if the first sort should be Descending, False if Ascending 
        4lbl2StringSecond label on which to sort. Optional. 
        5lbl2DescBooleanTrue if the second sort should be Descending, False if Ascending 
        6lbl3StringThird label on which to sort. Optional. 
        7lbl3DescBooleanTrue if the third sort should be Descending, False if Ascending 
        8delimStringThe delimiter used by label paths. If not provided, delim will default to "/" 
        	
         
        Code Sample
            Dim parser as New JsonParser()
            Dim dataobj as New JsonObject(true), sortedobj as JsonObject()
            Dim fpath as String
            fpath = CurDir() & "/myJSONData.json"
            Try
                Call parser.loadFromFile(fpath)
                Set dataObj = parser.getRootObject()
                If dataobj Is Nothing Then
                    Print "Error: Unable to parse file " & fpath
                    Exit Function
                End If
                Set sortedobj = jsonSort(dataObj, "company", False, "address/state", False, "address/city", False, "/")
            Catch
                Print "Error " & Error() & " on line " & Erl()
            End Try
